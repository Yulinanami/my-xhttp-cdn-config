
# XHTTP + CDN 搭建指南 (二)：Nginx配置与客户端模板

## 1. Nginx 反向代理配置

编辑 Nginx 配置文件（ `/etc/nginx/nginx.conf` ）

```bash
nano /etc/nginx/nginx.conf
```

清空初始配置，替换为下面的配置（你需要自行替换配置里面的占位配置）

```nginx
user root;
worker_processes auto;

error_log /usr/local/nginx/logs/error.log notice;
pid /run/nginx.pid;

events {
    worker_connections 1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    # 获取真实IP的设置
    set_real_ip_from      127.0.0.1;
    map $http_cf_connecting_ip $real_client_ip {
        default $http_cf_connecting_ip;
        ""      $remote_addr;
    }
    real_ip_header        X-Real-IP;

    # 优化参数
    sendfile              on;
    server_tokens         off;
    tcp_nodelay           on;
    tcp_nopush            on;
    client_max_body_size  0;
    gzip                  on;

    add_header X-Content-Type-Options nosniff;

    # SSL 通用配置
    ssl_session_cache          shared:SSL:16m;
    ssl_session_timeout        1h;
    ssl_session_tickets        off;
    ssl_protocols              TLSv1.3 TLSv1.2;
    ssl_ciphers                TLS_AES_256_GCM_SHA384:TLS_CHACHA20_POLY1305_SHA256:TLS_AES_128_GCM_SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers  on;
    ssl_stapling               on;
    ssl_stapling_verify        on;
    resolver                   1.1.1.1 8.8.8.8 valid=60s;
    resolver_timeout           2s;

    # 映射请求头
    map $real_client_ip $proxy_forwarded_elem {
        ~^[0-9.]+$        "for=$real_client_ip";
        ~^[0-9A-Fa-f:.]+$ "for=\"[$real_client_ip]\"";
        default           "for=unknown";
    }
    map $http_forwarded $proxy_add_forwarded {
        default "$proxy_forwarded_elem";
    }
    map $http_upgrade $connection_upgrade {
        default upgrade;
        ""      close;
    }

    server {
        listen       8003 ssl;
        http2        on;
        # Reality 的域名
        server_name  reality.yourdomain1.com;

        ssl_certificate /etc/ssl/private/fullchain.cer;
        ssl_certificate_key /etc/ssl/private/private.key;

        location / {
            # 建议填 VPS 所在地区的大学官网作为回落网站
            # 不建议填带路径的 URL
            proxy_pass https://www.stanford.edu;   # 你的 Reality 回落网站
            proxy_ssl_server_name on;
            proxy_ssl_name www.stanford.edu;
            proxy_redirect https://www.stanford.edu/ https://$host/;
            proxy_redirect http://www.stanford.edu/ https://$host/;
            proxy_set_header Host www.stanford.edu;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_set_header X-Forwarded-Host $host;
        }
    }

    server {
        listen       8003 ssl;
        http2        on;
        # CDN 的域名
        server_name  cdn.yourdomain2.com;

        ssl_certificate /etc/ssl/private/fullchain.cer;
        ssl_certificate_key /etc/ssl/private/private.key;

        location / {
            # 建议填 VPS 所在地区的大学官网作为回落网站
            proxy_pass https://www.harvard.edu;   # 你的 CDN 回落网站
            proxy_ssl_server_name on;
            proxy_ssl_name www.harvard.edu;
            proxy_redirect https://www.harvard.edu/ https://$host/;
            proxy_redirect http://www.harvard.edu/ https://$host/;
            proxy_set_header Host www.harvard.edu;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_set_header X-Forwarded-Host $host;
        }

        # 改成你的路径 /path
        location /cdn { 
            grpc_pass 127.0.0.1:8001;
            grpc_set_header Host                  $host;
            grpc_set_header X-Real-IP             $real_client_ip;
            grpc_set_header Forwarded             $proxy_add_forwarded;
            grpc_set_header X-Forwarded-For       $proxy_add_x_forwarded_for;
            grpc_set_header X-Forwarded-Proto     $scheme;
        }
    }

    server {
        listen  80 default_server;
       server_name _;

        return 301 https://$host$request_uri;
    }
}
```

注意：

- 回落网站建议使用 **大学官网首页 / 根站**，不要使用带路径的 URL。
- 例如 `https://www.u-tokyo.ac.jp/en/` 更推荐改成 `https://www.u-tokyo.ac.jp`。

```bash
# 测试nginx配置
nginx -t
```

## 2. Xray 服务端配置

获取并保持必要的配置参数
```bash
# 获取uuid1
xray uuid

# 获取uuid2
xray uuid

# 获取reality公私钥
xray x25519

# 生成 VLESS Encryption 密钥对 (防止 CDN 中间人解密流量)
xray vlessenc
# 输出的 encryption 填入客户端，decryption 填入服务端 XHTTP 入站
```

直接编辑并覆盖 `/usr/local/etc/xray/config.json` 文件：

```bash
nano /usr/local/etc/xray/config.json
```

```json
{
    "log": {
        "loglevel": "info"
    },
    "routing": {
        "domainStrategy": "IPIfNonMatch",
        "rules": [
            {
                "type": "field",
                "ip": [
                    "geoip:cn",
                    "geoip:private"
                ],
                "outboundTag": "block"
            }
        ]
    },
  "inbounds": [
    {
      "listen": "0.0.0.0",
      "port": 443,
      "protocol": "vless",
      "settings": {
        "clients": [
          {
             "id": "", // uuid1
            "level": 0,
            "flow": "xtls-rprx-vision"
          }
        ],
        "decryption": "none",
        "fallbacks": [
          {
            "dest": "8001",    //可以修改，保持和下面的port一致
            "xver": 0
          }
        ]
      },
      "streamSettings": {
        "network": "raw",
        "security": "reality",
        "realitySettings": {
          "show": false,
          "target": "8003",
          "xver": 0, 
          "serverNames": [
            "reality.yourdomain1.com"     //reality的域名
          ],
          "privateKey": "",    // 填写私钥
          "shortIds": [
              ""    // 0 to f, 必须双数，最长16位，可以留空
          ]
        },
              "sniffing": {
        "enabled": true,
        "destOverride": ["http", "tls", "quic"],
        "metadataOnly": false,
        "routeOnly": true
      }
      }
    },
    {
      "listen": "127.0.0.1",
      "port": 8001,   //和上面的dest保持一致
      "protocol": "vless",
      "settings": {
        "clients": [
          {
            "id": "",  // uuid2
            "level": 0
          }
        ],
        "decryption": ""  // 填写 xray vlessenc 生成的 decryption 值 (用于防止 CDN 中间人解密流量)，如无需加密填 "none"
      },
      "streamSettings": {
        "network": "xhttp",
        "xhttpSettings": {
          "host": "",   
          "path": "/cdn",    // 自行定义
          "mode": "auto"
        },
              "sniffing": {
        "enabled": true,
        "destOverride": ["http", "tls", "quic"],
        "metadataOnly": false,
        "routeOnly": true
      }
      }
    }
  ],
    "outbounds": [
        {
            "protocol": "freedom",
            "tag": "direct"
        },
        {
            "protocol": "blackhole",
            "tag": "block"
        }
    ]
}
```

> **关于 VLESS Encryption (vlessenc)**
>
> 当流量经过 CDN 时，CDN 会终结 TLS 再回源，理论上 CDN 可以看到解密后的明文。
> 启用 VLESS Encryption 后，Xray 会在 VLESS 协议层增加一层端到端加密（ML-KEM-768 + X25519 后量子安全算法），
> 即使 CDN 终结了 TLS，也无法解密流量内容。
>
> - 服务端 XHTTP 入站的 `decryption` 字段填入 `xray vlessenc` 生成的 decryption 值
> - 客户端对应节点的 `encryption` 字段填入生成的 encryption 值
> - 需要 Xray-core 较新版本支持
> - 仅 XHTTP 入站需要启用（因为只有它过 CDN），Vision 入站保持 `"none"` 即可

## 3. 测试与启动服务

配置文件修改完成后，执行以下命令测试配置可用性，并最终确保证书正确加载及服务正常运行：

```bash
# 测试 Nginx 配置
nginx -t

# 测试 xray 配置
xray -test -config /usr/local/etc/xray/config.json

#重启xray
systemctl restart xray && systemctl status xray

#重启nginx
systemctl restart nginx && systemctl status nginx
```

## 4. 客户端 (v2rayN) XHTTP Extra 模板

更推荐复制[客户端模板.txt](../客户端模板.txt)里面的模板，粘贴到V2rayN，然后在V2rayN里面替换占位符，这样会更方便一点

对照以下参数进行替换：
* `address`: 填入直连的服务器 IP 或 CDN 域名
* `serverName`: 填入你的 Reality 域名或 CDN 域名
* `publicKey`: 填入与服务端配对的公钥
* `path`: 填入你的伪装路径

### 模式三：上行 xhttp+TLS+CDN | 下行 xhttp+Reality
```json
{
  "downloadSettings": {
    "address": "YOUR_VPS_IP",
    "port": 443,
    "network": "xhttp",
    "security": "reality",
    "realitySettings": {
      "show": false,
      "serverName": "YOUR_REALITY_DOMAIN",
      "fingerprint": "chrome",
      "shortId": "YOUR_SHORT_ID",
      "publicKey": "YOUR_PUBILC_KEY"
    },
    "xhttpSettings": {
      "host": "",
      "path": "YOUR_PATH",
      "mode": "auto"
    }
  }
}
```

### 模式五：上行 xhttp+Reality | 下行 xhttp+TLS+CDN
```json
{
  "downloadSettings": {
    "address": "YOUR_CDN_DOMAIN",
    "port": 443,
    "network": "xhttp",
    "security": "tls",
    "tlsSettings": {
      "serverName": "YOUR_CDN_DOMAIN",
      "allowInsecure": false,
      "alpn": [
        "h2"
      ],
      "fingerprint": "chrome"
    },
    "xhttpSettings": {
      "host": "YOUR_CDN_DOMAIN",
      "path": "YOUR_PATH",
      "mode": "auto"
    }
  }
}
```
